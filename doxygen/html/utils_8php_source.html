436 /* DEPRECATED (use parseTable instead). Still used in form.php.
437  *
438  * Converts first count($keys) numerical indices to named keys (e.g. 2=>$value to $keys[2]=>$value),
439  * additional indices are changed to keys with value true (e.g. 42=>$value to $value=>true).
440  * So it's like php's array_combine(),
441  * but handling extra values and leaving existing named (associative) keys unchanged.
442  * Useful when you don't want to write the keys every time.
443  * Example: arrayToAssoc(array('myName','myValue','readonly','custom'=>'c'), array('name','value'))
444  * returns array('name'=>'myName', 'value'=>'myValue', 'custom'=>'c', 'readonly'=>true) */
445 function arrayToAssoc($array, $keys)
446 {
447  $result = $array;
448  for ($i = 0; $i < count($keys); $i++)
449  if (array_key_exists($i, $result))
450  {
451  $result[$keys[$i]] = $result[$i];
452  unset($result[$i]);
453  }
454  return $result;
455 }

457 /* Parses a table description into an associative array of associative arrays.
458  * Columns can have modifiers: t for translating (gettext), # for integers (intval), b for booleans.
459  * Example: for $description='
460  * NAME => TYPE; #VALUE;
461  * name1 => t1; 1v; custom => c1;
462  * name2 => t2; ; something => s1; readonly;
463  * name3 => t3;
464  * ')
465  * parseTable returns array(
466  * 'name1' => array('name'=>'name1', 'type'=>'t1', 'value'=>intval('1v'), 'custom'=>'c1'),
467  * 'name2' => array('name'=>'name2', 'type'=>'t2', 'value'=>'', 'something'=>s1', 'readonly'=>true),
468  * 'name3' => array('name'=>'name3', 'type'=>'t3')
469  * );
470  * The $gettext param is only used by the translator script to find strings to translate.
471  */
472 function parseTable($description, $gettext = 'gettext')
473 {
474  // Parse line by line.
475  $rows = explode("\n", $description);
476  $result = array();
477  $i = 0;
478  $headers = array();
479  $keyHeader = null;
480  foreach ($rows as $row)
481  {
482  // Skip empty lines.
483  $row = trim($row);
484  if (!strlen($row))
485  continue;
486  // Parse "key => field1; field2; field3;" (fields may contain '=>' but may not contain ';')
487  $row = explode('=>', $row, 2);
488  $key = trim($row[0]);
489  $fields = explode(';', $row[1]);
490  array_pop($fields); // Empty space after last ';'.
491  foreach ($fields as &$field)
492  $field = trim($field);
493  // First row describes key names (headers) and column modifiers.
494  if ($i == 0)
495  {
496  $keyHeader = allCapsToCamelCase($key);
497  foreach ($fields as &$h)
498  {
499  if ($h[0] == '#' || $h[0] == 'b' || $h[0] == 't')
500  {
501  $h = array('name' => substr($h, 1), 'modifier' => $h[0]);
502  }
503  else
504  $h = array('name' => $h, 'modifier' => ' ');
505  $h['name'] = allCapsToCamelCase($h['name']);
506  $headers[]= $h;
507  }
508  }
509  // Following rows are saved accordingly (fields are modified if header applies,
510  // split if they contain '=>').
511  else
512  {
513  $row = array();
514  foreach ($fields as $j => &$field)
515  {
516  if ($j < count($headers))
517  {
518  $f = trim($field);
519  switch ($headers[$j]['modifier'])
520  {
521  case '#': $f = intval($f); break;
522  case 'b': $f = $f === 'true' || $f === '1' | $f === 'True'; break;
523  case 't':
524  if (empty($f))
525  $f = '';
526  else if ($gettext == 'gettext')
527  $f = gettext($f);
528  else
529  call_user_func($gettext, $f, $headers[$j]['name'], $key); break;
530  }
531  $row[$headers[$j]['name']] = $f;
532  }
533  else if (strpos($field, '=>'))
534  {
535  $f = explode('=>', $field, 2);
536  $row[trim($f[0])] = trim($f[1]);
537  }
538  else
539  $row[trim($field)] = true;
540  }
541  $row[$keyHeader] = $key;
542  $result[$key] = $row;
543  }
544  $i++;
545  }
546  return $result;
547 }
