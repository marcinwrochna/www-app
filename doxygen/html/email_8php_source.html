158 /* Return true iff $email is a valid email address.
159  * by Douglas Lovell
160  * http://www.linuxjournal.com/article/9585?page=0,3 */
161 function validEmail($email)
162 {
163  $isValid = true;
164  $atIndex = strrpos($email, "@");
165  if (is_bool($atIndex) && !$atIndex) return false;
166 
167  $domain = substr($email, $atIndex+1);
168  $local = substr($email, 0, $atIndex);
169  $localLen = strlen($local);
170  $domainLen = strlen($domain);
171  if ($localLen < 1 || $localLen > 64) return false; // local part length exceeded
172  if ($domainLen < 1 || $domainLen > 255) return false; // domain part length exceeded
173  if ($local[0] == '.' || $local[$localLen-1] == '.') return false; // local part starts or ends with '.'
174  if (preg_match('/\\.\\./', $local)) return false; // local part has two consecutive dots
175  if (!preg_match('/^[A-Za-z0-9\\-\\.]+$/', $domain)) return false; // character not valid in domain part
176  if (preg_match('/\\.\\./', $domain)) return false; // domain part has two consecutive dots
177  if (!preg_match('/^(\\\\.|[A-Za-z0-9!#%&`_=\\/$\'*+?^{}|~.-])+$/',
178  str_replace("\\\\","",$local)))
179  {
180  // character not valid in local part unless
181  // local part is quoted
182  if (!preg_match('/^"(\\\\"|[^"])+"$/',
183  str_replace("\\\\","",$local))) return false;
184  }
185  if (DEBUG<1 && !(checkdnsrr($domain,"MX") || checkdnsrr($domain,"A"))) return false; // domain not found in DNS
186  return true;
187 }
